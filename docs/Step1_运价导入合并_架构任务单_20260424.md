### 0.2 数据流（Excel 三段式）

```
前端 <Excel tab> 拖入文件
  └─ POST /rate-batches/upload (FormData file + 可选 parser_hint)
      └─ rate_batch_service.create_draft_batch_from_upload
         ├─ 第 1 段：registry.resolve → adapter.parse → ParsedRateBatch → _draft_batches[bid]=draft
         │    └─ 返回 RateBatchDetail（source_type="excel", adapter_key 非空）
         ├─ 第 2 段：registry.resolve 抛 LookupError
         │    └─ _try_ai_fallback_on_excel(file_path, db)
         │         ├─ openpyxl 读所有 sheet 文本 → 拼成一坨纯文本 text
         │         ├─ parse_email_text(text, db) → legacy dict
         │         ├─ 若 len(parsed_rows) >= 1:
         │         │   ├─ legacy_payload_to_parsed_batch → ParsedRateBatch（adapter_key=None, file_type=ocean 作默认占位）
         │         │   ├─ _build_draft_batch(..., source_type="excel_ai_fallback") → _draft_batches[bid]=draft
         │         │   └─ 返回 RateBatchDetail（source_type="excel_ai_fallback"，前端据此展示"AI 解析"tag）
         │         └─ 若 len(parsed_rows) == 0: raise NoRatesFoundError
         └─ 第 3 段：NoRatesFoundError 冒泡
              └─ API 层返回 code=422 + 三语 key "upload.noRatesFound"（前端 message.warning 渲染）
```

      def _try_ai_fallback_on_excel(
          saved_path: Path, file_name: str, db: Session
      ) -> ParsedRateBatch:
          """第 2 段：读 Excel 纯文本 → parse_email_text → ParsedRateBatch。

          AI 抓不到任何行时抛 NoRatesFoundError（第 3 段）。
          """
          text = _extract_excel_text(saved_path)  # openpyxl 扫所有 sheet 所有非空单元格，按行拼接
          legacy = parse_email_text(text, db)     # 复用现有 AI 解析，返回 legacy dict
          if not legacy.get("parsed_rows"):
              raise NoRatesFoundError(
                  f"AI fallback found no rates in {file_name}"
              )
          # 强制 file_type 占位（Step1FileType 是枚举，不能为空）；激活器实际走 records 走，不依赖 file_type
          return legacy_payload_to_parsed_batch(
              legacy,
              file_type=Step1FileType.ocean,  # 占位，纯为通过 normalizer 校验
              adapter_key=None,
              source_file=file_name,
          )

      def _extract_excel_text(path: Path, max_chars: int = 20000) -> str:
          wb = openpyxl.load_workbook(path, data_only=True, read_only=True)
          lines: list[str] = []
          for sheet in wb.worksheets:
              lines.append(f"# Sheet: {sheet.title}")
              for row in sheet.iter_rows(values_only=True):
                  cells = [str(c) for c in row if c is not None and str(c).strip()]
                  if cells:
                      lines.append(" | ".join(cells))
              if sum(len(l) for l in lines) > max_chars:
                  break
          wb.close()
          return "\n".join(lines)[:max_chars]
